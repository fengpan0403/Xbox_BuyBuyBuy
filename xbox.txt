window.setInterval(function(){
	var buttonPanel = $("#buttonPanel_AppIdentityBuyButton");
	var iframes = $("iframe[class='pi-full-frame']");
	if(iframes.length == 0){
		buttonPanel.click();
	}else{
		var buttons = iframes.contents().find("button");
		if(buttons.length > 2) buttons[2].click();
		else if (buttons.length == 1) {
			var cancelButton = buttons[0];
			cancelButton.click();
		}
	}
}, 2000);
